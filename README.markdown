# My dot files && vim plugins

## Dot files included

* vimrc
* gvimrc
* bashrc
* bash_aliases
* irbrb
* railsrc

## Instalation

ln -s ~/.vim/vimrc ~/.vimrc
ln -s ~/.vim/gvimrc ~/.gvimrc
ln -s ~/.vim/bashrc ~/.bashrc
ln -s ~/.vim/bash_aliases ~/.bash_aliases
ln -s ~/.vim/irbrc ~/.irbrc
ln -s ~/.vim/railsrc ~/.railsrc
